/**
	Третья программа
	@алгоритм переворачивания строки
	P.S. алгоритм может понадобиться, допустим, для определения слов, которые читаются как слева направа, так и наоборот(палиндромы)
	Пример: "шалаш"
**/

string str;
main
{
	int size;
	size = 0;
	str = "а роза упала на лапу Азора";
	/**
		подсчёт количества символов в строке.
		Последний симлов '~'
	**/
	
	int i;
	i=0;
	while(elem(str,i)!='\n')				// можно было сразу воспользоваться функцией length
	{
		size = size+1;
	}
	
	string buf;
	int m;
	char a;
	/**
		переворачивание строки
	**/
	int j;
	j=size-1;
	for(; j>0; j=j-1)
	{
		a = elem(str,j);
		buf = buf+ToString(a);
		m = m+1;
	}
}
