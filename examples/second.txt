/**
	объявление глобальных переменных
**/

string str;
string s;
char b;
int kol;

/**
	глобальная функция
**/
int func(int j)
{
	int k;
	k = j+15;
	return k;
}

/**
	функция main
**/

main{
	// присваивание функции переменной
	int i = func(10);
	/*
	*	вывод на экран
	* 	Варианты, какие типы может принимать ф-я
	*	write(int numb);
	*	write(string str);
	*	write(char ch);
	*/
	write(i);
	
	// присваивание значения глобальным переменным
	b = 'b';
	s = "Hello world";
	
	write("Globals:");
	write(b);
	write(s);
	
	write("Введите строку:");
	// чтение строки с консоли
	string text;
	read(text);
	write(text); 
	
	text = "Hello "+"Java";
	
	if(contains(text, "Hello"))						// операция проверяет содержит ли данная строка указанную строку
	{
		write("Yes. Contains");
	}
	else{
	
		if(equals(str,"Hello"))					// операция проверяет равна ли данная строка указанной строке
		{
			int i1;
			i1 =14;
			write(i1);
		}
	}
	
	string hello_str = "Hello";
	/*
	*	equals(string firstStr, string secondStr)
	*/
	if(equals(hello_str,"Hello"))					// операция проверяет равна ли данная строка указанной строке
	{
		//int i1;			// нельзя объявить, т.к. переменная была объявлена выше, даже если в блоке if, она используется во всём блоке main
		i1 =14;
		i1 = 14+16;
		write("Yes. Equals!");
		write(i1);
	}
	
	if(!equals(s,"world") && !equals(s,"Hello"))	// && - обозначает "и". Знак "||" - обазначает или, знак ! - обозначает не
	{
		write("No equals");
	}
	
	string hg;
	hg = "strABC";
	
	/*
	*	вариант использования оператора if
	*	можно сравнивать любые одинаковые типы данных(т.е. int, string, char)
	*	оператора сравнения, которые можно использовать: >,<,>=,<=,==,!=
	*/
	if(hg=="strABC")		
	{
		write("Good");
	}
	
	/*
	*	ещё о встроенных функциях
	*/
	
	/*
	*	subString(string str, int index)
	*	или
	*	subString(string str, int indexBegin, int indexEnd)
	*/
	string subStr;
	subStr = subString(hg, 2);				// результат - подстрока ABC. поиск осуществляется от заданного элемента до конца строки.
	subStr = subString(hg, 2, 4);			// результат - подстрока AB. указывается индекс начала и конца.
	
	/*
	*	replace(string str, char a, char b)
	*	replaceFirst(string str, string A, string B)
	*/
	subStr = replace(hg,'A', 'B');			//заменяет в строке все A на B
	subStr = replaceFirst(hg, "A", "B");		//заменяет в строке первый найденный элемент
	
	/**
		цикл for
		@ elem(string str, int i) - функция, преобразовывабщая i-ы элемент в char
		  length(string str) - функция, определяющая длину строки
	**/
	
	int k;
	k=0;
	int j = 0;
	char ch;
	ch = elem(text,j);
	write(ch);
	
	for(j=0; j<length(text); j=j+1)
	{
		if(contains(text, "l"))			//подсчёт сколько букв в в строке "Hello world"
		{
			k = k+1;
		}
	}
	
	write(k);
	
	/**
	*	цикл while
	**/
	str = "acdnqqqqn";
	i=0;
	while(elem(str,i)!='n')
	{
		char v;
		v = elem(str,i);
		string v_str;
		v_str = ToString(v);
		write(v_str);
		i=i+1;
	}
	write("KOLICHESTVO:");
	write(i);
	
	write("Arifmetics:");
	int sum = k+13;
	int sum1;
	write(sum);
	/*
	*	от числа 3 отнимаем -3
	*	ВНИМАНИЕ. если выполнить данный код kol=3-3, тогда работать не будет, не явно опередела операция разности,
	*	поэтому необходимо данный код записывать так kol=3 - 3, для явного опеделения в правиле, разделяя прабелами числа
	*/
	kol = 3 - -3;						
	sum1 = kol+func(k);
	write(sum1);
	sum = sum-sum1;
	write(sum);
}
